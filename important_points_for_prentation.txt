           ↓

           ↓
    Results Display (web/templates/results.html)

           METHODOLOGY

STEP 2: Text Preprocessing
├── Lowercase conversion

├── Remove stopwords (a, an, the, etc.)

STEP 3: Feature Extraction
├── TF-IDF Vectorization
└── Convert text to numerical vectors

STEP 4: Skill Extraction
├── 200+ skill database
├── Pattern matching & NER
└── Extract skills from both documents

         MATHEMATICAL MODEL
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. TEXT SIMILARITY (Cosine Similarity)
   
   Similarity(A,B) = (A·B) / (||A|| × ||B||)
   
   Where:
   A, B = TF-IDF vectors of resume and job
   Range: 0 (different) to 1 (identical)

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

2. SKILL MATCH SCORE
   
   Skill_Score = |R ∩ J| / |J|
   
   Where:
   R = Skills in Resume
   J = Skills in Job Description
   Range: 0 to 1

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

3. FINAL WEIGHTED SCORE
   
   Match_Score = (Wt × Text_Sim) + (Ws × Skill_Score)
   
   Where:
   Wt = Text weight (0.6)
   Ws = Skill weight (0.4)
   Final Score × 100 = Percentage (0-100%)

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

EXAMPLE CALCULATION:
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
If Text_Sim = 0.85 and Skill_Score = 0.75
Then Match_Score = (0.6×0.85) + (0.4×0.75) = 0.81
Final = 81% Match
